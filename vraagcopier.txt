// ==UserScript==
// @name         text copier woods 1.3
// @namespace    http://tampermonkey.net/
// @version      2026-05-22
// @description  Copy question + 3 answers
// @author       toxivid
// @match        https://app.woots.nl/*
// @grant        GM_setClipboard
// ==/UserScript==

(function () {
    'use strict';

    if (window.__copyPromptBound) {
        console.log("Already active");
        return;
    }

    window.__copyPromptBound = true;

    document.addEventListener("keydown", (e) => {

        // Ctrl + Alt + C
        if (e.ctrlKey && e.altKey && e.key.toLowerCase() === "c") {

            console.log("Ctrl+Alt+C detected");

            // Pak de vraag div
            const subquestion = document.querySelector(".quiz-subquestion");

            if (!subquestion) {
                console.log("Question div not found");
                return;
            }

            // Alle redactor-content elementen binnen deze vraag
            const texts = subquestion.querySelectorAll(".redactor-content");

            console.log("Found:", texts.length);

            // Minimaal vraag + 3 antwoorden
            if (texts.length < 4) {
                console.log("Not enough text elements");
                return;
            }

            // Vraag
            const question = texts[0].innerText.trim();

            // Antwoorden
            const answer1 = texts[1].innerText.trim();
            const answer2 = texts[2].innerText.trim();
            const answer3 = texts[3].innerText.trim();

            // Eind tekst
            const finalText =
`QUESTION:
${question}

ANSWERS:
A) ${answer1}
B) ${answer2}
C) ${answer3}`;

            // Copy
            GM_setClipboard(finalText);

            console.log("Copied:\n", finalText);
        }
    });

    console.log("Ctrl+Alt+C copy enabled");
})();
